最简单的方法：使用Django命令

  # 激活虚拟环境并进入后端目录
  source .venv/Scripts/activate
  cd CampusFeedbackSystem/backend

  # 生成5个超级管理员邀请码
  python manage.py generate_invitation_codes --role super_admin --count 5

  # 生成自定义邀请码
  python manage.py generate_invitation_codes --role super_admin --custom-code MYCODE123

  # 生成内容管理员邀请码
  python manage.py generate_invitation_codes --role content_admin --count 3
